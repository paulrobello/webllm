```cpp
// Mount the buffer in MEMFS at /tmp/webllm-model.gguf and load via llama_model_load_from_file.
// Returns model handle on success, nullptr on failure.
// The buffer is COPIED into MEMFS — caller may free `buf` after this returns.
void* webllm_load_model(const void* buf, size_t n_bytes) {
    const char* path = "/tmp/webllm-model.gguf";
    FILE* f = std::fopen(path, "wb");
    if (!f) return nullptr;
    size_t wrote = std::fwrite(buf, 1, n_bytes, f);
    std::fclose(f);
    if (wrote != n_bytes) return nullptr;

    llama_model_params mparams = llama_model_default_params();
    // n_gpu_layers default is 0; we want all layers on the (only) backend.
    // ggml-webgpu registers itself as the only available backend, so layers
    // routed to it regardless of this setting under Emscripten.
    mparams.n_gpu_layers = 999;
    return llama_model_load_from_file(path, mparams);
}

void webllm_free_model(void* model) {
    if (model) llama_model_free(static_cast<llama_model*>(model));
}
```

```cpp
// Create a llama_context for the given model.
// n_ctx: KV cache size in tokens. 0 = use model's training-time default.
// embeddings: 0 = causal LM (logits), 1 = embedder mode.
// pooling_type: 0=NONE, 1=MEAN, 2=CLS, 3=LAST. Ignored when embeddings=0.
// flash_attn: 0/1.
void* webllm_create_context(void* model, int32_t n_ctx, int32_t embeddings,
                            int32_t pooling_type, int32_t flash_attn) {
    if (!model) return nullptr;
    llama_context_params cparams = llama_context_default_params();
    if (n_ctx > 0) cparams.n_ctx = (uint32_t) n_ctx;
    cparams.embeddings = embeddings != 0;
    cparams.pooling_type = (enum llama_pooling_type) pooling_type;
    cparams.flash_attn = flash_attn != 0;
    // Single-thread under Emscripten — pthreads not enabled in this build.
    cparams.n_threads = 1;
    cparams.n_threads_batch = 1;
    return llama_init_from_model(static_cast<llama_model*>(model), cparams);
}

void webllm_free_context(void* ctx) {
    if (ctx) llama_free(static_cast<llama_context*>(ctx));
}
```

- [ ] **Step 1: Implement decode + logits read**

Append:
```cpp
// Decode n_tokens tokens at sequence positions [past_len, past_len+n_tokens).
// All tokens go to seq_id=0. Logits are computed for the LAST token only
// (sufficient for greedy / single-step decode; P5 spec-decode will need
// a richer logits-mask API).
// Returns 0 on success, llama.cpp non-zero status on failure.
int32_t webllm_decode(void* ctx, const int32_t* token_ids, int32_t n_tokens,
                      int32_t past_len) {
    if (!ctx || n_tokens <= 0) return -1;
    llama_context* lctx = static_cast<llama_context*>(ctx);

    llama_batch batch = llama_batch_init(n_tokens, /*embd*/ 0, /*n_seq_max*/ 1);
    for (int32_t i = 0; i < n_tokens; ++i) {
        batch.token[i] = token_ids[i];
        batch.pos[i] = past_len + i;
        batch.n_seq_id[i] = 1;
        batch.seq_id[i][0] = 0;
        batch.logits[i] = (i == n_tokens - 1) ? 1 : 0;
    }
    batch.n_tokens = n_tokens;

    int32_t status = llama_decode(lctx, batch);
    llama_batch_free(batch);
    return status;
}

// Returns pointer to logits for the i-th token of the last decode batch.
// ith=-1 → last logits-flagged token. Pointer is into ctx-owned memory;
// valid until the next decode call.
const float* webllm_get_logits(void* ctx, int32_t ith) {
    if (!ctx) return nullptr;
    llama_context* lctx = static_cast<llama_context*>(ctx);
    return ith < 0 ? llama_get_logits(lctx) : llama_get_logits_ith(lctx, ith);
}
```

```typescript
// SPDX-License-Identifier: MIT
// Tier 3 P0 spike — bindings for webllm_* bridge exports wrapping
// upstream llama.cpp's llama_model / llama_context / llama_decode.
// See docs/superpowers/specs/2026-05-05-tier3-llama-decode-migration-design.md.

export interface LlamaBridge {
	loadModel(buf: Uint8Array): Promise<bigint>;
	freeModel(handle: bigint): void;
	createContext(model: bigint, params: {
		nCtx: number;
		embeddings?: boolean;
		poolingType?: 0 | 1 | 2 | 3;
		flashAttn?: boolean;
	}): bigint;
	freeContext(ctx: bigint): void;
	decode(ctx: bigint, tokens: Int32Array, pastLen: number): Promise<number>;
	getLogits(ctx: bigint, model: bigint, ith?: number): Float32Array;
	nVocab(model: bigint): number;
}

interface RawModule {
	_webllm_load_model: (bufPtr: bigint, nBytes: bigint) => bigint;
	_webllm_free_model: (handle: bigint) => void;
	_webllm_create_context: (
		model: bigint,
		nCtx: number,
		embeddings: number,
		poolingType: number,
		flashAttn: number,
	) => bigint;
	_webllm_free_context: (ctx: bigint) => void;
	_webllm_decode: (
		ctx: bigint,
		tokensPtr: bigint,
		nTokens: number,
		pastLen: number,
	) => Promise<number>;
	_webllm_get_logits: (ctx: bigint, ith: number) => bigint;
	_webllm_n_vocab: (model: bigint) => number;
	_bridge_malloc: (size: bigint) => bigint;
	_bridge_free: (ptr: bigint) => void;
	HEAPU8: Uint8Array;
	HEAPF32: Float32Array;
	HEAP32: Int32Array;
	Asyncify?: { handleAsync: <T>(fn: () => Promise<T>) => Promise<T> };
}

export function createLlamaBridge(mod: RawModule): LlamaBridge {
	return {
		async loadModel(buf: Uint8Array): Promise<bigint> {
			const ptr = mod._bridge_malloc(BigInt(buf.byteLength));
			if (ptr === 0n) throw new Error("webllm: bridge_malloc failed");
			try {
				mod.HEAPU8.set(buf, Number(ptr));
				const handle = mod._webllm_load_model(ptr, BigInt(buf.byteLength));
				if (handle === 0n) throw new Error("webllm: webllm_load_model returned null");
				return handle;
			} finally {
				mod._bridge_free(ptr);
			}
		},
		freeModel(handle: bigint): void {
			mod._webllm_free_model(handle);
		},
		createContext(model, params) {
			const handle = mod._webllm_create_context(
				model,
				params.nCtx,
				params.embeddings ? 1 : 0,
				params.poolingType ?? 0,
				params.flashAttn ? 1 : 0,
			);
			if (handle === 0n) throw new Error("webllm: webllm_create_context returned null");
			return handle;
		},
		freeContext(ctx: bigint): void {
			mod._webllm_free_context(ctx);
		},
		async decode(ctx, tokens, pastLen) {
			const nBytes = BigInt(tokens.byteLength);
			const ptr = mod._bridge_malloc(nBytes);
			if (ptr === 0n) throw new Error("webllm: bridge_malloc failed");
			try {
				new Int32Array(
					mod.HEAPU8.buffer,
					Number(ptr),
					tokens.length,
				).set(tokens);
				const status = await mod._webllm_decode(ctx, ptr, tokens.length, pastLen);
				return status;
			} finally {
				mod._bridge_free(ptr);
			}
		},
		getLogits(ctx, model, ith = -1) {
			const ptr = mod._webllm_get_logits(ctx, ith);
			if (ptr === 0n) throw new Error("webllm: webllm_get_logits returned null");
			const nVocab = mod._webllm_n_vocab(model);
			return new Float32Array(mod.HEAPU8.buffer, Number(ptr), nVocab);
		},
		nVocab(model: bigint): number {
			return mod._webllm_n_vocab(model);
		},
	};
}
```

```javascript
// Replace these with values from PROMPT-FIXTURE.md (Task 0 Step 1)
const PROMPT_TOKEN_IDS = [/* fill from fixture */];
const EXPECTED_PARIS_ID = /* fill from fixture */;

export async function runSpike() {
	try {
		log("[1/6] Initializing WASM module...");
		const mod = await createModule();
		const bridge = createLlamaBridge(mod);

		log("[2/6] Initializing WebGPU backend...");
		await mod._webgpu_init();

		log("[3/6] Fetching TinyLlama Q4_0 GGUF...");
		// Path served by `make smoke-serve`; adjust if your server roots elsewhere.
		const ggufUrl = "/tinyllama-1.1b-chat-v1.0.Q4_0.gguf";
		const buf = new Uint8Array(await (await fetch(ggufUrl)).arrayBuffer());
		log(`     loaded ${(buf.byteLength / 1024 / 1024).toFixed(1)} MiB`);

		const model = await bridge.loadModel(buf);
		const ctx = bridge.createContext(model, { nCtx: 512 });
		log(`     vocab = ${bridge.nVocab(model)}`);

		log("[5/6] Decoding prompt...");
		const promptTokens = new Int32Array(PROMPT_TOKEN_IDS);
		const status = await bridge.decode(ctx, promptTokens, 0);
		if (status !== 0) {
			log(`     llama_decode returned status ${status}`, "fail");
			return;
		}

		log("[6/6] Reading logits + argmax...");
		const logits = bridge.getLogits(ctx, model);
		let topId = 0;
		let topVal = -Infinity;
		for (let i = 0; i < logits.length; i++) {
			if (logits[i] > topVal) { topVal = logits[i]; topId = i; }
		}
		log(`     top-1 token id = ${topId} (logit ${topVal.toFixed(3)})`);

		if (topId === EXPECTED_PARIS_ID) {
			log("PASS — top-1 matches \" Paris\"", "pass");
		} else {
			log(`FAIL — expected ${EXPECTED_PARIS_ID}, got ${topId}`, "fail");
		}

		bridge.freeContext(ctx);
		bridge.freeModel(model);
	} catch (err) {
		log(`FAIL — ${err.message}\n${err.stack ?? ""}`, "fail");
	}
}
```
